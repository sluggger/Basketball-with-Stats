from bokeh.plotting import figure, output_notebook, show
from bokeh.models import HoverTool, ColumnDataSource

output_notebook()

x = stats["MIN"]
y = stats["PTS"]
name = stats["Name"]

source = ColumnDataSource(data=dict(x=x, y=y, name=name))

fig = figure(
    tools="pan, box_zoom, wheel_zoom, zoom_in, zoom_out, reset, save",
    title="Example Bokeh plot",
    y_range=[0, 2000],
    x_axis_label='Minutes',
    y_axis_label='Points',
    width=600,
    height=400,
)

scatter = fig.circle('x', 'y', source=source, legend_label="y=x", fill_color="white", size=4)
hover_tool = HoverTool(renderers=[scatter], tooltips=[('Name', '@name')])
fig.add_tools(hover_tool)

show(fig)

import plotly.graph_objects as go

x = stats["3PA"]
y = stats["3PM"]
name = stats["Name"]  # Additional value to show on hover

fig = go.Figure(
    layout=dict(
        title="Shooters gonna shoot",
        yaxis_range=[0, 300],
        xaxis=dict(title='3PM'),
        yaxis=dict(title='3PA'),
    ),
)

# Add hovertemplate to show 'Name' value on hover
scatter = go.Scatter(x=x, y=y, mode='markers', name="y=x", marker=dict(color='purple', size=8),
                     hovertemplate='Name: %{text}<br>3PA: %{x}<br>3PM: %{y}')
fig.add_trace(scatter)

fig.update_traces(text=name)  # Set 'Name' as the hover text

fig.show()
